# Preloader 方式 — DaemonSet 预下载模型到 HostPath

通过 DaemonSet 在每个 GPU 节点上预先下载模型文件到 HostPath，Pod 启动时只需挂载节点目录并等待 `.ready` 文件就绪即可。

---

### 1. Model Preloader DaemonSet

```yaml
# model-preloader-daemonset.yaml
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: model-preloader
  namespace: llm-inference
  labels:
    app: model-preloader
spec:
  selector:
    matchLabels:
      app: model-preloader
  template:
    metadata:
      labels:
        app: model-preloader
    spec:
      affinity:
        nodeAffinity:
          requiredDuringSchedulingIgnoredDuringExecution:
            nodeSelectorTerms:
            - matchExpressions:
              - key: nvidia.com/gpu.present
                operator: In
                values:
                - "true"
      initContainers:
      - name: model-puller
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          set -e

          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY

          MODEL_DIR=/data/models/qwen3-06b
          READY_FILE=$MODEL_DIR/.ready

          # ========== 幂等性检查：已存在则跳过 ==========
          if [ -f "$READY_FILE" ]; then
            echo "[$(date)] 模型已存在于 HostPath，跳过下载。"
            echo "[$(date)] 本地大小: $(du -sh $MODEL_DIR | cut -f1)"
            echo "[$(date)] 文件列表:"
            ls -lh $MODEL_DIR/*.safetensors 2>/dev/null | head -5
            exit 0
          fi

          echo "[$(date)] 首次同步：从 MinIO 下载 llm-models/Qwen3-06b ..."
          mkdir -p $MODEL_DIR

          # 从 MinIO 增量同步到 HostPath 挂载路径
          mc mirror --overwrite --remove \
            models/llm-models/Qwen3-0.6B/ \
            $MODEL_DIR/

          # 写入就绪标志（关键：后续重建秒级跳过）
          touch $READY_FILE
          echo "[$(date)] 同步完成。总大小: $(du -sh $MODEL_DIR | cut -f1)"

          # 关键文件校验
          for FILE in config.json tokenizer.json; do
            if [ ! -f "$MODEL_DIR/$FILE" ]; then
              echo "[$(date)] 关键文件缺失: $FILE"
              exit 1
            fi
          done
          echo "[$(date)] 关键文件校验通过"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "2Gi"
            cpu: "2"
          requests:
            memory: "512Mi"
            cpu: "500m"
      containers:
      # Preloader 完成下载后，由 pause 容器保持 Pod Running
      - name: pause
        image: registry.k8s.io/pause:3.9
        resources:
          limits:
            memory: "64Mi"
            cpu: "100m"
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
```

### 2. vLLM Deployment（挂载 HostPath）

```yaml
# vllm-deployment-hostpath.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: vllm-qwen3-06b
  namespace: llm-inference
  labels:
    app: vllm-qwen3-06b
spec:
  replicas: 1
  selector:
    matchLabels:
      app: vllm-qwen3-06b
  template:
    metadata:
      labels:
        app: vllm-qwen3-06b
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8000"
        prometheus.io/path: "/metrics"
    spec:
      nodeSelector:
        kubernetes.io/hostname: 10-23-106-50   # 节点选择，一般用 nvidia.com/gpu.present=true
      initContainers:
      - name: model-ready-check
        image: busybox:1.36
        command:
        - sh
        - -c
        - |
          for i in $(seq 1 300); do
            if [ -f /data/models/qwen3-06b/.ready ]; then
              echo "[$(date)] Model ready file detected. Proceeding to vLLM startup."
              exit 0
            fi
            echo "[$(date)] Waiting for model preloader on this node... ($i/300)"
            sleep 2
          done
          echo "[$(date)] Timeout: model preloader did not complete on this node"
          exit 1
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "64Mi"
            cpu: "100m"
      containers:
      - name: vllm
        image: vllm/vllm-openai:v0.11.2
        args:
          - /data/models/qwen3-06b
          - --served-model-name
          - qwen3-06b
          - --dtype
          - half
          - --tensor-parallel-size
          - "1"
          - --max-model-len
          - "4096"
          - --gpu-memory-utilization
          - "0.8"
          - --host
          - 0.0.0.0
          - --port
          - "8000"
          - --uvicorn-log-level
          - info
        ports:
        - containerPort: 8000
          name: http
          protocol: TCP
        resources:
          limits:
            nvidia.com/gpu: "1"
            memory: "16Gi"
            cpu: "4"
          requests:
            nvidia.com/gpu: "1"
            memory: "16Gi"
            cpu: "4"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
          readOnly: true
        livenessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 300
          periodSeconds: 30
          timeoutSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 120
          periodSeconds: 15
          timeoutSeconds: 5
        startupProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 60
          periodSeconds: 10
          failureThreshold: 60
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
---
apiVersion: v1
kind: Service
metadata:
  name: vllm-service
  namespace: llm-inference
spec:
  type: ClusterIP
  selector:
    app: vllm-qwen3-06b
  ports:
  - port: 8000
    targetPort: 8000
    name: http
```
