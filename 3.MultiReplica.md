## Preloader方式
直接将模型文件下载到节点上，Pod直接挂载节点目录上的资源。

部署完成后就可以测试了。

预下载模型：
```bash
cat >> model-preloader-daemonset.yaml <<EOF
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: model-preloader
  namespace: llm-inference
  labels:
    app: model-preloader
spec:
  selector:
    matchLabels:
      app: model-preloader
  template:
    metadata:
      labels:
        app: model-preloader
    spec:
      affinity:
        nodeAffinity:
          requiredDuringSchedulingIgnoredDuringExecution:
            nodeSelectorTerms:
            - matchExpressions:
              - key: nvidia.com/gpu.present
                operator: In
                values:
                - "true"
      initContainers:
      - name: model-puller
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          set -e

          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY

          MODEL_DIR=/data/models/qwen3-0.6B
          READY_FILE=$MODEL_DIR/.ready

          # ========== 幂等性检查：已存在则跳过 ==========
          if [ -f "$READY_FILE" ]; then
            echo "[$(date)] 模型已存在于 HostPath，跳过下载。"
            echo "[$(date)] 本地大小: $(du -sh $MODEL_DIR | cut -f1)"
            echo "[$(date)] 文件列表:"
            ls -lh $MODEL_DIR/*.safetensors 2>/dev/null | head -5
            exit 0
          fi

          echo "[$(date)] 首次同步：从MinIO下载 llm-models/Qwen3-0.6B ..."
          mkdir -p $MODEL_DIR

          # 从MinIO增量同步到HostPath挂载路径
          mc mirror --overwrite --remove \
            models/llm-models/Qwen3-0.6B/ \
            $MODEL_DIR/

          # 写入就绪标志（关键：后续重建秒级跳过）
          touch $READY_FILE
          echo "[$(date)] 同步完成。总大小: $(du -sh $MODEL_DIR | cut -f1)"

          # 关键文件校验
          for FILE in config.json tokenizer.json; do
            if [ ! -f "$MODEL_DIR/$FILE" ]; then
              echo "[$(date)] 关键文件缺失: $FILE"
              exit 1
            fi
          done
          echo "[$(date)] 关键文件校验通过"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "2Gi"
            cpu: "2"
          requests:
            memory: "512Mi"
            cpu: "500m"
      containers:
      - name: model-watcher
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY

          MODEL_DIR=/data/models/qwen3-0.6B
          CHECK_INTERVAL=60

          echo "[$(date)] 模型监控已启动，检查间隔 ${CHECK_INTERVAL}s"

          while true; do
            missing=false

            for file in config.json tokenizer.json; do
              if [ ! -s "$MODEL_DIR/$file" ]; then
                echo "[$(date)] 缺失或为空的文件: $file"
                missing=true
                break
              fi
            done

            if ! ls "$MODEL_DIR"/*.safetensors >/dev/null 2>&1; then
              echo "[$(date)] 未找到 .safetensors 权重文件"
              missing=true
            fi

            if [ "$missing" = true ]; then
              echo "[$(date)] 检测到模型不完整，开始从 MinIO 修复..."
              if mc mirror --overwrite --remove \
                   models/llm-models/Qwen3-0.6B/ \
                   "$MODEL_DIR/"; then
                touch $READY_FILE
                echo "[$(date)] 修复完成"
                touch $MODEL_DIR/.ready
              else
                echo "[$(date)] 修复失败，将在下一轮重试"
              fi
            else
              echo "[$(date)] 模型文件完整"
            fi

            sleep $CHECK_INTERVAL
          done
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "1Gi"
            cpu: "1"
          requests:
            memory: "256Mi"
            cpu: "250m"
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
EOF
```

vllm-ingress.yaml
```bash
cat >> vllm-ingress.yaml <<EOF
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: vllm-ingress
  namespace: llm-inference
  annotations:
    # ========== Least-Conn：适配 LLM 长连接流式输出 ==========
    nginx.ingress.kubernetes.io/load-balance-algorithm: "least_conn"
    nginx.ingress.kubernetes.io/proxy-read-timeout: "600"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "600"
    nginx.ingress.kubernetes.io/proxy-connect-timeout: "600"
    nginx.ingress.kubernetes.io/proxy-body-size: "50m"
    # 保持长连接，减少 TCP 握手开销
    nginx.ingress.kubernetes.io/upstream-keepalive-connections: "32"
spec:
  ingressClassName: nginx
  rules:
  - host: vllm.demo.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: vllm-service
            port:
              number: 8000
EOF
```

vllm-statefulset.yaml
```bash
cat >> vllm-statefulset.yaml <<EOF
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: vllm-qwen3-06b
  namespace: llm-inference
  labels:
    app: vllm-qwen3-06b
spec:
  serviceName: vllm-headless
  replicas: 2
  podManagementPolicy: Parallel
  selector:
    matchLabels:
      app: vllm-qwen3-06b
  template:
    metadata:
      labels:
        app: vllm-qwen3-06b
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8000"
        prometheus.io/path: "/metrics"
    spec:
      # ========== Pod 反亲和性：强制跨节点分散 ==========
      affinity:
        podAntiAffinity:
          requiredDuringSchedulingIgnoredDuringExecution:
          - labelSelector:
              matchExpressions:
              - key: app
                operator: In
                values:
                - vllm-qwen3-06b
            topologyKey: kubernetes.io/hostname
      # 轻量就绪检查：依赖步骤 5 的 DaemonSet Preloader
      initContainers:
      - name: model-ready-check
        image: busybox:1.36
        command:
        - sh
        - -c
        - |
          for i in $(seq 1 300); do
            if [ -f /data/models/qwen3-0.6B/.ready ]; then
              echo "[$(date)] Model ready file detected on this node. Proceeding."
              exit 0
            fi
            echo "[$(date)] Waiting for model preloader... ($i/300)"
            sleep 2
          done
          echo "[$(date)] Timeout: model preloader not ready on this node"
          exit 1
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "64Mi"
            cpu: "100m"
      containers:
      - name: vllm
        image: vllm/vllm-openai:v0.11.2
        args:
          - /data/models/qwen3-0.6B
          - --served-model-name
          - qwen3-06b
          - --dtype
          - half
          - --tensor-parallel-size
          - "1"
          - --max-model-len
          - "4096"
          - --gpu-memory-utilization
          - "0.9"
          - --host
          - 0.0.0.0
          - --port
          - "8000"
          - --uvicorn-log-level
          - info
        ports:
        - containerPort: 8000
          name: http
          protocol: TCP
        resources:
          limits:
            nvidia.com/gpu: "1"
            memory: "12Gi"
            cpu: "4"
          requests:
            nvidia.com/gpu: "1"
            memory: "6Gi"
            cpu: "4"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
          readOnly: true
        livenessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 300
          periodSeconds: 30
          timeoutSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 120
          periodSeconds: 15
          timeoutSeconds: 5
        startupProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 60
          periodSeconds: 10
          failureThreshold: 60
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
EOF
```
