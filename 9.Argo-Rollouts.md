```yaml
# vllm-rollout.yaml
apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: vllm-qwen3-06b
  namespace: llm-inference
  labels:
    app: vllm-qwen3-06b
spec:
  replicas: 3
  selector:
    matchLabels:
      app: vllm-qwen3-06b
  strategy:
    canary:
      canaryService: vllm-canary
      stableService: vllm-stable
      # 关键：限制 canary 同时运行的最大数量，防止超量占用 GPU
      maxSurge: "25%"        # 基于 replicas 的百分比，3*25% = 0.75 → 向上取整为 1
      maxUnavailable: 1      # 允许 1 个 stable Pod 不可用，确保 canary 能调度
      trafficRouting:
        nginx:
          stableIngress: vllm-stable
          #annotationPrefix: nginx.ingress.kubernetes.io
      steps:
      # Step 1: 25% 流量 → 1 canary / (2 stable + 1 canary) = 1/3 ≈ 33%，但权重设为 25
      - setWeight: 25
      - pause: {duration: 3m}
      # Step 2: Prometheus 自动验收
      - analysis:
          templates:
          - templateName: vllm-prometheus-smoke
      # Step 3: 50% 流量 → 需要 2 canary / (1 stable + 2 canary)
      - setWeight: 50
      - pause: {duration: 5m}
      # Step 4: 二次验收
      - analysis:
          templates:
          - templateName: vllm-prometheus-smoke
      # Step 5: 全量切换（100% 新版本）
      - setWeight: 100
  template:
    metadata:
      labels:
        app: vllm-qwen3-06b
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8000"
        prometheus.io/path: "/metrics"
    spec:
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
          - weight: 100
            podAffinityTerm:
              labelSelector:
                matchExpressions:
                - key: app
                  operator: In
                  values:
                  - vllm-qwen3-06b
              topologyKey: kubernetes.io/hostname
      initContainers:
      - name: model-ready-check
        image: busybox:1.36
        command:
        - sh
        - -c
        - |
          for i in $(seq 1 300); do
            if [ -f /data/models/qwen3-0.6B/.ready ]; then
              echo "[$(date)] Model ready file detected. Proceeding."
              exit 0
            fi
            echo "[$(date)] Waiting for model preloader... ($i/300)"
            sleep 2
          done
          echo "[$(date)] Timeout: model preloader not ready on this node"
          exit 1
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "64Mi"
            cpu: "100m"
      containers:
      - name: vllm
        image: vllm/vllm-openai:v0.11.2
        args:
          - /data/models/qwen3-0.6B
          - --served-model-name
          - qwen3-06b
          # - --quantization 
          # - "awq"
          - --tensor-parallel-size
          - "1"
          - --max-model-len
          - "2048"
          - --max-num-batched-tokens
          - "4096"
          - --max-num-seqs
          - "8"
          - --enable-chunked-prefill
          - --enable-prefix-caching
          - --gpu-memory-utilization
          - "0.45"
          - --host
          - 0.0.0.0
          - --port
          - "8000"
          - --uvicorn-log-level
          - info
        ports:
        - containerPort: 8000
          name: http
          protocol: TCP
        resources:
          limits:
            nvidia.com/gpu: "1"
            memory: "16Gi"
            cpu: "4"
          requests:
            nvidia.com/gpu: "1"
            memory: "8Gi"
            cpu: "4"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
          readOnly: true
        livenessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 300
          periodSeconds: 30
          timeoutSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 120
          periodSeconds: 15
          timeoutSeconds: 5
        startupProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 60
          periodSeconds: 10
          failureThreshold: 60
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
```
