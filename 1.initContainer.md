#### 4. vllm-deployment.yaml

```bash
cat >> vllm-deployment.yaml <<EOF
apiVersion: apps/v1
kind: Deployment
metadata:
  name: vllm-qwen3-06b
  namespace: llm-inference
  labels:
    app: vllm-qwen3-06b
spec:
  replicas: 1
  selector:
    matchLabels:
      app: vllm-qwen3-06b
  template:
    metadata:
      labels:
        app: vllm-qwen3-06b
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8000"
        prometheus.io/path: "/metrics"
    spec:
      # 固定到 node01，与 LocalPV 同节点，避免跨节点挂载失败
      nodeSelector:
        kubernetes.io/hostname: k8s-node01
      initContainers:
      - name: model-puller
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          set -e
          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY
          MODEL_DIR=/data/models/qwen3-06b
          mkdir -p $MODEL_DIR
          # 从MinIO增量同步到PVC挂载路径
          mc mirror --overwrite --remove models/llm-models/Qwen3-0.6B/ $MODEL_DIR/
          ls -al $MODEL_DIR
        volumeMounts:
        - name: model-storage
          mountPath: /data/models
        resources:
          limits:
            memory: "2Gi"
            cpu: "2"
          requests:
            memory: "512Mi"
            cpu: "500m"
      containers:
        - name: vllm
          #image: vllm/vllm-openai:latest
          image: vllm/vllm-openai:v0.11.2
          args:
            - /data/models/qwen3-06b
            - --served-model-name
            - qwen3-06b
            - --dtype
            - bfloat16
            - --tensor-parallel-size
            - "1"
            - --max-model-len
            - "4096"
            - --gpu-memory-utilization
            - "0.9"
            - --host
            - 0.0.0.0
            - --port
            - "8000"
            - --uvicorn-log-level
            - info
          ports:
            - containerPort: 8000
              name: http
              protocol: TCP
          resources:
            limits:
              nvidia.com/gpu: "1"
              memory: "16Gi"
              cpu: "4"
            requests:
              nvidia.com/gpu: "1"
              memory: "8Gi"
              cpu: "4"
          volumeMounts:
            - name: model-storage
              mountPath: /data/models
              readOnly: true
          livenessProbe:
            httpGet:
              path: /health
              port: 8000
            initialDelaySeconds: 300
            periodSeconds: 30
            timeoutSeconds: 10
          readinessProbe:
            httpGet:
              path: /health
              port: 8000
            initialDelaySeconds: 120
            periodSeconds: 15
            timeoutSeconds: 5
          startupProbe:
            httpGet:
              path: /health
              port: 8000
            initialDelaySeconds: 60
            periodSeconds: 10
            failureThreshold: 60
      volumes:
        - name: model-storage
          persistentVolumeClaim:
            claimName: qwen3-06b-model-pvc-vllm
---
apiVersion: v1
kind: Service
metadata:
  name: vllm-service
  namespace: llm-inference
spec:
  type: ClusterIP
  selector:
    app: vllm-qwen3-06b
  ports:
    - port: 8000
      targetPort: 8000
      name: http
---
EOF
```

#### vLLM

**ingress：**

```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: openwebui-ingress
  namespace: default               # 根据实际命名空间修改
  annotations:
    # 兼容旧版 Ingress 资源指定 Ingress Class（保留）
    kubernetes.io/ingress.class: nginx

    # 上传文件 / 大请求体支持（设置为 0 表示不限制，可按需调整为具体大小）
    nginx.ingress.kubernetes.io/proxy-body-size: "0"

    # 长连接超时，适配 WebSocket 及长时间推理请求
    nginx.ingress.kubernetes.io/proxy-read-timeout: "3600"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "3600"

    # 若使用 cert-manager 自动签发 TLS 证书，可取消下行注释
    # cert-manager.io/cluster-issuer: "letsencrypt-prod"

spec:
  ingressClassName: nginx
  rules:
  - host: openwebui.demo.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: open-webui-service
            port:
              number: 8080         # Open WebUI 默认服务端口
```
