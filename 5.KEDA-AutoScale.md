keda-scaledobject.yaml
```yaml
apiVersion: keda.sh/v1alpha1
kind: ScaledObject
metadata:
  name: vllm-qwen3-06b-scaler
  namespace: llm-inference
spec:
  scaleTargetRef:
    name: vllm-qwen3-06b
    kind: StatefulSet
    apiVersion: apps/v1
  pollingInterval: 15
  cooldownPeriod: 300
  minReplicaCount: 1
  maxReplicaCount: 2  # 硬上限：与 2×RTX 3090 对齐
  # ========== Fallback：Prometheus 异常时的安全策略 ==========
  fallback:
    failureThreshold: 3
    replicas: 1
  advanced:
    horizontalPodAutoscalerConfig:
      behavior:
        scaleUp:
          stabilizationWindowSeconds: 30
          policies:
          - type: Pods
            value: 1
            periodSeconds: 10
          selectPolicy: Max
        scaleDown:
          stabilizationWindowSeconds: 120
          policies:
          - type: Pods
            value: 1
            periodSeconds: 30 
          selectPolicy: Min
  triggers:
  # ========== 触发器 1：队列深度（主触发器，最可靠） ==========
  - type: prometheus
    metadata:
      serverAddress: http://prometheus-server.monitoring.svc.cluster.local:80
      metricName: vllm-num-requests-waiting
      query: |
        sum(vllm:num_requests_waiting{namespace="llm-inference"})
      threshold: "2"
  # ========== 触发器 2：P99 TTFT（延迟质量兜底） ==========
  - type: prometheus
    metadata:
      serverAddress: http://prometheus-server.monitoring.svc.cluster.local:80
      metricName: vllm-p99-ttft-ms
      query: |
        histogram_quantile(0.99, sum(rate(vllm:time_to_first_token_seconds_bucket{namespace="llm-inference"}[2m])) by (le)) * 1000
      threshold: "600"
```

model-preloader-daemonset.yaml
```yaml
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: model-preloader
  namespace: llm-inference
  labels:
    app: model-preloader
spec:
  selector:
    matchLabels:
      app: model-preloader
  template:
    metadata:
      labels:
        app: model-preloader
    spec:
      affinity:
        nodeAffinity:
          requiredDuringSchedulingIgnoredDuringExecution:
            nodeSelectorTerms:
            - matchExpressions:
              - key: kubernetes.io/hostname
                operator: In
                values:
                - k8s-node01.magedu.com
                - k8s-node03.magedu.com
      initContainers:
      - name: model-puller
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          set -e

          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY

          MODEL_DIR=/data/models/qwen3-0.6B
          READY_FILE=$MODEL_DIR/.ready

          # ========== 幂等性检查：已存在则跳过 ==========
          if [ -f "$READY_FILE" ]; then
            echo "[$(date)] 模型已存在于 HostPath，跳过下载。"
            echo "[$(date)] 本地大小: $(du -sh $MODEL_DIR | cut -f1)"
            echo "[$(date)] 文件列表:"
            ls -lh $MODEL_DIR/*.safetensors 2>/dev/null | head -5
            exit 0
          fi

          echo "[$(date)] 首次同步：从MinIO下载 llm-models/Qwen3-0.6B ..."
          mkdir -p $MODEL_DIR

          # 从MinIO增量同步到HostPath挂载路径
          mc mirror --overwrite --remove \
            models/llm-models/Qwen3-0.6B/ \
            $MODEL_DIR/

          # 写入就绪标志（关键：后续重建秒级跳过）
          touch $READY_FILE
          echo "[$(date)] 同步完成。总大小: $(du -sh $MODEL_DIR | cut -f1)"

          # 关键文件校验
          for FILE in config.json tokenizer.json; do
            if [ ! -f "$MODEL_DIR/$FILE" ]; then
              echo "[$(date)] 关键文件缺失: $FILE"
              exit 1
            fi
          done
          echo "[$(date)] 关键文件校验通过"
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "2Gi"
            cpu: "2"
          requests:
            memory: "512Mi"
            cpu: "500m"
      containers:
      - name: model-watcher
        image: minio/mc:RELEASE.2025-08-13T08-35-41Z
        env:
        - name: MINIO_ENDPOINT
          value: "http://minio.minio.svc.cluster.local:9000"
        - name: MINIO_ACCESS_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_ACCESS_KEY
        - name: MINIO_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: minio-credentials
              key: MINIO_SECRET_KEY
        command:
        - sh
        - -c
        - |
          # 配置 MinIO 别名
          mc alias set models $MINIO_ENDPOINT $MINIO_ACCESS_KEY $MINIO_SECRET_KEY

          MODEL_DIR=/data/models/qwen3-0.6B
          CHECK_INTERVAL=60

          echo "[$(date)] 模型监控已启动，检查间隔 ${CHECK_INTERVAL}s"

          while true; do
            missing=false

            for file in config.json tokenizer.json; do
              if [ ! -s "$MODEL_DIR/$file" ]; then
                echo "[$(date)] 缺失或为空的文件: $file"
                missing=true
                break
              fi
            done

            if ! ls "$MODEL_DIR"/*.safetensors >/dev/null 2>&1; then
              echo "[$(date)] 未找到 .safetensors 权重文件"
              missing=true
            fi

            if [ "$missing" = true ]; then
              echo "[$(date)] 检测到模型不完整，开始从 MinIO 修复..."
              if mc mirror --overwrite --remove \
                   models/llm-models/Qwen3-0.6B/ \
                   "$MODEL_DIR/"; then
                touch $READY_FILE
                echo "[$(date)] 修复完成"
              else
                echo "[$(date)] 修复失败，将在下一轮重试"
              fi
            else
              echo "[$(date)] 模型文件完整"
            fi

            sleep $CHECK_INTERVAL
          done
        volumeMounts:
        - name: host-model-cache
          mountPath: /data/models
        resources:
          limits:
            memory: "1Gi"
            cpu: "1"
          requests:
            memory: "256Mi"
            cpu: "250m"
      volumes:
      - name: host-model-cache
        hostPath:
          path: /data/models
          type: DirectoryOrCreate
```

### 下面使用KEDA进行自动扩缩容

```bash
# keda-scaledobject.yaml
......
  triggers:
  # ========== 触发器 1：队列深度（主触发器，最可靠） ==========
  - type: prometheus
    metadata:
      serverAddress: http://prometheus-server.monitoring.svc.cluster.local:80 # prometheus服务地址
      metricName: vllm-num-requests-waiting                       # 指标中找这个指标
      query: |
        sum(vllm:num_requests_waiting{namespace="llm-inference"}) # 这是值
      threshold: "2"                                              # 一段时间内等待数值大于2则扩容
  # ========== 触发器 2：P99 TTFT（延迟质量兜底） ==========
  - type: prometheus
    metadata:
      serverAddress: http://prometheus-server.monitoring.svc.cluster.local:80
      metricName: vllm-p99-ttft-ms
      query: |
        histogram_quantile(0.99, sum(rate(vllm:time_to_first_token_seconds_bucket{namespace="llm-inference"}[2m])) by (le)) * 1000
      threshold: "600" # 延迟大于600ms则扩容

# 两个语句为或关系
```
